# - bg_rye_farms: Continental

# - bg_maize_farms: Temperate (Preferable South)
# - bg_millet_farms: Desert
# - bg_crystal_corn_farms: Crystal Empire
#
# - bg_livestock_ranches: Every state
#
# - bg_coffee_plantations: Tropical
# - bg_cotton_plantations: Temperate-Tropical, No Islands
# - bg_silk_plantations: Temperate, East Coast
# - bg_dye_plantations: Tropical, place together with coffee for now
# - bg_opium_plantations: Placement Criteria Undecided
# - bg_tea_plantations: Temperate-Tropical, East Coast
# - bg_tobacco_plantations: Worldwide
# - bg_sugar_plantations: Tropical-Temperate, Coast
# - bg_apple_plantations: Temperate
# - bg_banana_plantations: Tropical
# - bg_mango_plantations: Tropical
#

#
