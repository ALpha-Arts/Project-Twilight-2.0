decision_switch_flag = {
	is_shown = {
		exists = c:EQS
		this = c:EQS
		game_date < 1837
	}

	possible = {
		game_date < 1837
	}

	when_taken = {
		if = {
			limit = {
				has_variable = old_flag
			}
			custom_tooltip = {
				text = "decision_switch_flag_use_new_flag_tt"
				remove_variable = old_flag
			}
		}
		else = {
			custom_tooltip = {
				text = "decision_switch_flag_use_old_flag_tt"
				set_variable = old_flag
			}
		}
	}

	ai_chance = {
		base = 0
	}
}
