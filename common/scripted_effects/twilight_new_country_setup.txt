# Generic Setup #

effect_new_country_setup = {
	# Religions #

	if = {
		limit = {
			religion = { has_discrimination_trait = harmony }
		}
		ig:ig_devout = {
			set_interest_group_name = ig_philiologists
			remove_ideology = ideology_pious
			remove_ideology = ideology_moralist
			effect_remove_gender_ideologies = yes
			add_ideology = ideology_friendship_moralist
			add_ideology = ideology_feminist_ig
			add_ideology = ideology_anti_slavery
		}
	}

	# Gender Restrictions #

	# Only trigger if gender ideology was not manually assigned to ig_landowners
	if = {
		limit = {
			ig:ig_landowners = { has_gender_ideology = no }
		}
		# Failsafe: If the society is both patriarchal and matriarchal, then it's neither
		if = {
			limit = {
				AND = {
					country_is_patriarchal_society = yes
					country_is_matriarchal_society = yes
				}
			}
			ig:ig_landowners = { add_ideology = ideology_traditionalist_ig }
		}
		else_if = {
			limit = { country_is_patriarchal_society = yes }
			# If equality has already been achieved, make the landowners be moderate
			if = {
				limit = { has_law = law_type:law_womens_suffrage }
				ig:ig_landowners = { add_ideology = ideology_patriarchal_suffrage }
			}
			else = {
				ig:ig_landowners = { add_ideology = ideology_patriarchal }
			}
		}
		else_if = {
			limit = { country_is_matriarchal_society = yes }
			# If equality has already been achieved, make the landowners be moderate
			if = {
				limit = { has_law = law_type:law_womens_suffrage }
				ig:ig_landowners = { add_ideology = ideology_matriarchal_suffrage }
			}
			else = {
				ig:ig_landowners = { add_ideology = ideology_matriarchal }
			}
		}
		# Fallback
		else = {
			ig:ig_landowners = { add_ideology = ideology_traditionalist_ig }
		}
	}

	# Replace politicians with the "lesser" gender
	# The specific ratios should follow
	# the female_politician_chance and commander_leader_chance of the IGs
	# Patriarchy
	if = {
		limit = { has_law = law_type:law_no_womens_rights }
		every_interest_group = {
			leader = {
				if = {
					limit = { is_female = yes }
					# 0% of this gender, 0% of the total
					kill_character = yes
				}
			}
		}
	}
	else_if = {
		limit = { has_law = law_type:law_women_own_property }
		every_interest_group = {
			leader = {
				if = {
					limit = { is_female = yes }
					# 20% of this gender, 10% of the total
					random_list = {
						80 = { kill_character = yes }
						20 = {}
					}
				}
			}
		}
	}
	else_if = {
		limit = { has_law = law_type:law_women_in_the_workplace }
		every_interest_group = {
			leader = {
				if = {
					limit = { is_female = yes }
					# 40% of this gender, 20% of the total
					random_list = {
						60 = { kill_character = yes }
						40 = {}
					}
				}
			}
		}
	}
	# Matriarchy
	if = {
		limit = { has_law = law_type:law_no_mens_rights }
		every_interest_group = {
			leader = {
				if = {
					limit = { is_female = no }
					# 0% of this gender, 0% of the total
					kill_character = yes
				}
			}
		}
	}
	else_if = {
		limit = { has_law = law_type:law_men_own_property }
		every_interest_group = {
			leader = {
				if = {
					limit = { is_female = no }
					# 20% of this gender, 10% of the total
					random_list = {
						80 = { kill_character = yes }
						20 = {}
					}
				}
			}
		}
	}
	else_if = {
		limit = { has_law = law_type:law_men_in_the_workplace }
		every_interest_group = {
			leader = {
				if = {
					limit = { is_female = no }
					# 40% of this gender, 20% of the total
					random_list = {
						60 = { kill_character = yes }
						40 = {}
					}
				}
			}
		}
	}

	# Slavery #

	if = {
		limit = {
			NOT = { has_law = law_type:law_slavery_banned }
		}
		ig:ig_landowners = { add_ideology = ideology_pro_slavery }
	}

	# Variables #

	if = {
		limit = {
			country_is_alicornarchy = yes
			NOT = { has_variable = alicornarchy }
		}
		set_variable = alicornarchy
	}
}
