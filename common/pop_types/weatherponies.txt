weatherponies = {
	texture = "gfx/interface/icons/pops_icons/machinists.dds"
	color = hsv{ 0.60 0.90 0.50 }
	strata = middle
	start_quality_of_life = 10
	wage_weight = 1.5
	literacy_target = 0.20
	dependent_wage = 0.5	# per year
	unemployment = yes

	# 50% - 100% politically engaged
	# The same as engineers
	political_engagement_base = 0.5
	political_engagement_literacy_factor = 1.0

	political_engagement_mult = {
		value = 1.0

		add = {
			desc = "POP_STARVATION"

			if = {
				limit = {
					standard_of_living < 5
				}
				value = 1
			}
		}

		multiply = {
			desc = "NO_LABOR_MOVEMENT"
			value = 1.0

			if = {
				limit = {
					owner = {
						NOT = { has_technology_researched = labor_movement }
					}
				}
				value = 0.5
			}
		}

		multiply = {
			desc = "HAS_POLITICAL_AGITATION"
			value = 1.0

			if = {
				limit = {
					owner = {
						has_technology_researched = political_agitation
					}
				}
				value = 1.5
			}
		}
	}

	qualifications_growth_desc = "WEATHERPONIES_QUALIFICATIONS_DESC"
	qualifications = {
		# baseline: literacy * 20, norm 10 for Laborers with Literacy 25%
		# Requires weather magic
		if = {
			limit = {
				culture = {
					has_discrimination_trait = weather_magic
				}
			}

			add = {
				value = literacy_rate
				multiply = 20
				desc = "QUALIFICATIONS_LITERACY_FACTOR"
			}

			if = {
				limit = {
					OR = {
						is_pop_type = flightponies
						is_pop_type = laborers
					}
				}
				multiply = {
					value = 2
					desc = "QUALIFICATIONS_FAVORED_TYPE"
				}
			}

			if = {
				limit = {
					is_peasant_under_serfdom = yes
				}
				multiply = {
					value = 0.1
					desc = "QUALIFICATIONS_PEASANTS_UNDER_SERFDOM"
				}
			}

			multiply = {
				value = 1
				desc = "QUALIFICATIONS_WEATHER_MAGIC_FACTOR"
			}
		}
		else = {
			subtract = {
				value = 3334
				desc = "QUALIFICATIONS_NO_WEATHER_MAGIC_FACTOR"
			}
		}
	}

	portrait_age = {
		integer_range = {
			min = define:NPortrait|GRACEFUL_AGING_START
			max = define:NPortrait|GRACEFUL_AGING_END
		}
	}
	portrait_pose = { value = 0 }
	portrait_is_female = { always = yes }
}
