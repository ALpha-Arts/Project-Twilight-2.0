# Fallback party for Devout
# When devout are powerful, other religious IG's might join

religious_party = {
	color = { 100 100 100 }

	name = {
		first_valid = {
			triggered_desc = {
				desc = party_dance
				trigger = {
					exists = ig:ig_devout
					exists = py:religious_party
					ig:ig_devout = {
						is_member_of_party = root.py:religious_party
						leader = {
							has_variable = is_pinkie
						}
					}
				}
			}
			triggered_desc = {
				desc = party_friendship
				trigger = {
					country_has_state_religion = rel:magic_of_friendship
				}
			}
			triggered_desc = {
				desc = party_religious
				trigger = {
					always = yes
				}
			}
		}
	}

	icon = {
		default = "/gfx/interface/icons/political_parties/religious.dds"
	}

	# THIS IS RUN ONLY ONCE AT GAME START
	valid_for_country = {
		always = yes
	}

	available_for_interest_group = {
		owner = {
			country_has_voting_franchise = yes
		}
	}

	# Root - IG considering joining
	# scope:number - the number of IGs that would join this party before the root IG
	join_weight = {
		value = {
			value = -5
			desc = "base_value"
		}

		if = {
			limit = {
				exists = owner.ig:ig_landowners
				owner = {
					# If the landowners are relevant,
					ig:ig_landowners = {
						is_marginal = no
					}
					# and the state religion is not philosophical
					religion = {
						NOT = {
							has_discrimination_trait = philosophical
						}
					}
				}
			}
			add = {
				value = -5
				desc = "from_relevant_landowners"
			}
		}

		if = {
			limit = {
				is_interest_group_type = ig_devout
				owner = {
					has_law = law_type:law_theocracy
				}
			}
			add = {
				value = 5
				desc = "from_theocracy"
			}
		}

		# Nobody will join if the devout don't join
		if = {
			limit = {
				NOT = {
					is_interest_group_type = ig_devout
				}
				OR = {
					NOT = {
						exists = owner.py:religious_party
					}
					AND = {
						exists = owner.py:religious_party
						owner = {
							ig:ig_devout = {
								NOT = {
									is_member_of_party = owner.py:religious_party
								}
							}
						}
					}
				}
			}
			add = {
				value = -20
				# This should never be shown in game
				# because the party won't exist
				desc = "from_no_devout_in_party"
			}
		}

		# A party for powerful religious people
		if = {
			limit = {
				is_interest_group_type = ig_devout
				is_powerful = yes
			}
			add = {
				value = 10
				desc = "from_powerful_devout"
			}
		}
		else_if = {
			limit = {
				is_interest_group_type = ig_devout
				is_powerful = no
			}
			add = {
				value = 5
				desc = "from_influential_devout"
			}
		}

		# Twilight and her friends
		if = {
			limit = {
				exists = owner.ig:ig_devout.leader
				owner.ig:ig_devout.leader = {
					character_is_twilights_friend = yes
				}
				leader = {
					character_is_twilights_friend = yes
				}
			}
			add = {
				value = 5
				desc = "from_leader_is_twilights_friend"
			}
		}

		# If the devout are powerful, other influential IGs might join
		if = {
			limit = {
				is_powerful = no
				owner = {
					ig:ig_devout = {
						is_powerful = yes
					}
				}
			}
			# If they are religiously inclined,
			if = {
				limit = {
					leader = {
						has_ideology = ideology_theocrat
					}
				}
				add = {
					value = 10
					desc = "from_powerful_devout_other"
				}
			}
			# Or the state religion is Friendship
			else_if = {
				limit = {
					owner = {
						country_has_state_religion = rel:magic_of_friendship
					}
				}
				add = {
					value = 5
					desc = "from_powerful_devout_other_friendship"
				}
			}
		}

		# From Total Seperation Law
		if = {
			limit = {
				OR = {
					is_interest_group_type = ig_devout
					leader = {
						has_ideology = ideology_theocrat
					}
				}
				owner = {
					has_law = law_type:law_total_separation
				}
			}
			add = {
				value = 10
				desc = "from_total_separation"
			}
		}

		# Anti-religious characters and IGs will never join
		# unless the state religion is philosophical
		if = {
			limit = {
				OR = {
					has_ideology = ideology_anti_clerical
					leader = {
						has_ideology = ideology_atheist
					}
				}
				owner.religion = {
					NOT = {
						has_discrimination_trait = philosophical
					}
				}
			}
			add = {
				value = -15
				desc = "from_antireligious"
			}
		}

		# Patriarchs and matriarchs more likely
		# to join conservatives with gender equality
		if = {
			limit = {
				OR = {
					AND = {
						has_ideology = ideology_patriarchal
						owner = {
							country_is_patriarchal_society = yes
						}
					}
					AND = {
						has_ideology = ideology_matriarchal
						owner = {
							country_is_matriarchal_society = yes
						}
					}
				}
				owner = {
					has_law = law_type:law_womens_suffrage
				}
			}
			add = {
				value = 3
				desc = "from_patriarchy_or_matriarchy"
			}
		}

		if = {
			limit = {
				is_interest_group_type = ig_devout
				is_powerful = yes
			}
			multiply = {
				value = 2
				desc = "from_archetype_party"
			}
		}
		if = {
			limit = {
				scope:number = 1
				is_powerful = no
				is_marginal = no
			}
			multiply = {
				value = 2
				desc = "from_influential_supporting_member"
			}
		}
		else_if = {
			limit = {
				scope:number = 2
			}
			multiply = {
				value = 0.5
				desc = "from_two_prior_members"
			}
		}
		else_if = {
			limit = {
				scope:number = 3
			}
			multiply = {
				value = 0.25
				desc = "from_three_prior_members"
			}
		}
		else_if = {
			limit = {
				scope:number > 3
			}
			multiply = {
				value = 0.1
				desc = "from_more_than_three_prior_members"
			}
		}
	}
}
