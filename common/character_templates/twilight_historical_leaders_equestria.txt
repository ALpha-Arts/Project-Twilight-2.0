	}
	interest_group_leader_usage = {
		country_trigger = {
			c:EQS ?= THIS

			}
		}
		interest_group_trigger = {
			# is_marginal = no
			OR = {
				is_interest_group_type = ig_devout
				is_interest_group_type = ig_rural_folk
			}
		}
		earliest_usage_date = 1836.1.2
		latest_usage_date = 1890.1.1
		chance = 50
	}
	on_created = {
		set_variable = is_pinkie
		set_variable = plot_armor
		# TODO: Add Cheese Sandwich
		# 20s
		if = {
			limit = {
				game_date < 1842.1.1
			}
			# 20% chance to have been married
			random_list = {
				1 = {
					set_variable = is_married
				}
				4 = {}
			}
		}
		# 30s
		if = {
			limit = {
				game_date >= 1842.1.1
			}
			# 50% chance to have been married
			random_list = {
				1 = {
					set_variable = is_married
				}
				1 = {}
			}
		}
		# 40s
		else_if = {
			limit = {
				game_date >= 1852.1.1
			}
			# 80% chance to have been married
			random_list = {
				4 = {
					set_variable = is_married
				}
				1 = {}
			}
		}
	}
}
