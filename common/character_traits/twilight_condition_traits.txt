# Vanilla Traits #

alcoholic = {
	type = condition
	texture = "gfx/interface/icons/character_trait_icons/alcoholic.dds"

	character_modifier = {
		character_health_add = -0.1
		character_popularity_add = -10
	}

	command_modifier = {
		unit_morale_damage_mult = -0.1
	}

	country_modifier = {
		country_influence_mult = 0.02
	}

	interest_group_modifier = {
	}

	possible = {
		age > define:NPortrait|GRACEFUL_AGING_START
	}

	weight = {
		value = 20
		if = {
			limit = {
				OR = {
					has_variable = plot_armor
					has_addiction = yes
					has_trait = cautious
				}
			}
			multiply = 0.1
		}
	}

	replace = {
	}
}
