# State geography triggers #

state_is_in_equus = {
	OR = {
		region = sr:region_equestria
		region = sr:region_undiscovered_west
		region = sr:region_frozen_north
		region = sr:region_southern_equus
		region = sr:region_forbidden_jungle
		region = sr:region_saddle_arabia
		region = sr:region_changeling_lands
		region = sr:region_bone_dry_desert
	}
}

# Country geography triggers #
# exists = capital checks are necessary only because we may test these triggers
# and scope switch to capitals before states are initialized properly

country_is_in_equus = {
	exists = capital
	capital = {
		state_is_in_equus = yes
	}
}
