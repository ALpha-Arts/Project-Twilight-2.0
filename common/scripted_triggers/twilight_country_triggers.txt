	exists = ruler
	ruler = {
		has_variable = is_twilight
	}
	NOT = {
		has_law = law_type:law_universal_suffrage
	}
}

# scope:target_country is country
country_is_cats_vs_dogs = {
	has_strategic_adjacency = scope:target_country
	OR = {
		AND = {
			country_has_primary_culture = cu:cat
			scope:target_country = {
				country_has_primary_culture = cu:diamond_dog
			}
			cu:diamond_dog = {
				culture_is_discriminated_in = PREV
			}
		}
		AND = {
			country_has_primary_culture = cu:diamond_dog
			scope:target_country = {
				country_has_primary_culture = cu:cat
			}
			cu:cat = {
				culture_is_discriminated_in = PREV
			}
		}
	}
}
