je_trans_equestrian_railway = {
	icon = "gfx/interface/icons/event_icons/event_industry.dds"

	is_shown_when_inactive = {
		is_player = yes
		NOT = {
			has_global_variable = trans_equestrian_done
		}
		any_state = {
			filter = {
				OR = {
					state_region = s:STATE_MANEHATTAN
					state_region = s:STATE_CANTERLOT
					state_region = s:STATE_LAS_PEGASUS
				}
			}
			percent > 0.8
			owner = ROOT
		}
	}

	possible = {
		has_technology_researched = railways
	}

	complete = {
		any_scope_state = {
			state_region = s:STATE_MANEHATTAN
			any_scope_building = {
				is_building_type = building_railway
				OR = {
					has_active_production_method = pm_wooden_passenger_carriages
					has_active_production_method = pm_steel_passenger_carriages
				}
				level >= 5
			}
		}
		any_scope_state = {
			state_region = s:STATE_CANTERLOT
			any_scope_building = {
				is_building_type = building_railway
				OR = {
					has_active_production_method = pm_wooden_passenger_carriages
					has_active_production_method = pm_steel_passenger_carriages
				}
				level >= 5
			}
		}
		any_scope_state = {
			state_region = s:STATE_LAS_PEGASUS
			any_scope_building = {
				is_building_type = building_railway
				OR = {
					has_active_production_method = pm_wooden_passenger_carriages
					has_active_production_method = pm_steel_passenger_carriages
				}
				level >= 5
			}
		}
	}

	on_complete = {
		set_global_variable = trans_equestrian_done
		trigger_event = {
			id = twilight_major_railways.1
		}
	}

	weight = 50
	inheritable = no
}
